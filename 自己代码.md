def a(x):
    n=1
    while int(x/10)!= 0:
        n=n+1
        x=int(x/10)
    return n

i=int(input("输入待检测数字：" ))
c=i
sum=0
b=1
n=a(i)
while (c>0):
    j=c%10
    while n>0:
        b=b*j
        n=n-1
    sum=sum+b
    c=c/10
if sum==i:
    print("该数字是阿姆斯特朗数")
else:
    print("该数字不是阿姆斯特朗数")
